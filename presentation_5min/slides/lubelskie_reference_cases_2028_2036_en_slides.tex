\slide{How to Apply to Lubelskie (2028--2036 Draft)}{
\textbf{ERDF layer}
\begin{itemize}
\item Priority: infrastructure, mobility access, low-carbon operational upgrades.
\item Indicative band: \textbf{EUR 28--57m} per package.
\end{itemize}

\textbf{JTF layer}
\begin{itemize}
\item Priority: vulnerable groups in transition (low-income households, maintenance and mobility workers).
\item Indicative band: \textbf{EUR 14--26m} per package.
\end{itemize}

\textbf{Combined design rule}
\begin{itemize}
\item One regional package, two funds, clear sequencing from ERDF to JTF, no budget overlap.
\end{itemize}
